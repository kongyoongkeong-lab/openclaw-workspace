"""
CSV Analyzer Python Script
Analyzes CSV files to calculate mean, median, and mode for specified columns.
"""

import pandas as pd
import statistics
from typing import Union


def load_csv(filepath: str) -> pd.DataFrame:
    """
    Read a CSV file and return a DataFrame.
    
    Args:
        filepath: Path to the CSV file
        
    Returns:
        pandas DataFrame
        
    Raises:
        FileNotFoundError: If the file does not exist
    """
    try:
        return pd.read_csv(filepath)
    except FileNotFoundError:
        raise FileNotFoundError(f"File not found: {filepath}")
    except Exception as e:
        raise ValueError(f"Error reading CSV file: {e}")


def calculate_mean(series: pd.Series) -> float:
    """
    Calculate the arithmetic mean of a series of values.
    
    Args:
        series: pandas Series of numeric values
        
    Returns:
        Mean value (float)
    """
    return float(statistics.mean(series))


def calculate_median(series: pd.Series) -> float:
    """
    Calculate the median of a series of values.
    
    Args:
        series: pandas Series of numeric values
        
    Returns:
        Median value (float)
    """
    return float(statistics.median(series))


def calculate_mode(series: pd.Series) -> Union[float, None]:
    """
    Calculate the mode (most frequent value) of a series of values.
    
    Args:
        series: pandas Series of values
        
    Returns:
        Mode value (float or None)
    """
    value_counts = series.value_counts()
    if len(value_counts) == 0:
        return None
    if len(value_counts) == 1:
        return float(value_counts.iloc[0])
    return value_counts.idxmax()


def analyze_csv(filepath: str, 
                columns: list) -> str:
    """
    Analyze a CSV file for mean, median, and mode statistics.
    
    Args:
        filepath: Path to the CSV file
        columns: List of column names to analyze
        
    Returns:
        Formatted string with statistics
    """
    try:
        df = load_csv(filepath)
    except FileNotFoundError as e:
        return f"ERROR: {e}"
    except ValueError as e:
        return f"ERROR: {e}"
    
    if not columns:
        return "ERROR: No columns specified"
    
    results = []
    for col in columns:
        if col not in df.columns:
            results.append(f"ERROR: Column '{col}' not found")
            continue
        
        col_data = df[col]
        
        try:
            mean_val = calculate_mean(col_data)
            median_val = calculate_median(col_data)
            mode_val = calculate_mode(col_data)
            
            results.append(f"### Mean")
            results.append(f"- **{col}**: {mean_val}")
            
            results.append(f"### Median")
            results.append(f"- **{col}**: {median_val}")
            
            if mode_val is not None:
                results.append(f"### Mode")
                results.append(f"- **{col}**: {mode_val}")
            else:
                results.append(f"### Mode")
                results.append(f"- **{col}**: None (no unique values)")
                
        except (TypeError, ZeroDivisionError) as e:
            results.append(f"ERROR: {e}")
    
    return "\n\n".join(results)
